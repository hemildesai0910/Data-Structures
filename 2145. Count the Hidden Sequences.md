#include <vector>
#include <algorithm>
using namespace std;

class Solution {
public:
  int numberOfArrays(vector<int>& differences, int lower, int upper) {
      int cur = 0;
      int minval = 0;
      int maxval = 0;

      for (int &a : differences) {
          cur += a;
          minval = min(minval, cur);
          maxval = max(maxval, cur);
          if ((upper - maxval) - (lower - minval) + 1 <= 0) {
              return 0;
          }
      }
      return (upper - maxval) - (lower - minval) + 1;
  }
};
